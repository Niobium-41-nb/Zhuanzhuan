% ==================== 9. 认证与安全 ====================
\section{认证与安全}

\subsection{JWT 认证流程}

\begin{enumerate}[itemsep=5pt]
    \item \textbf{登录：} 用户提交凭证，服务端验证后生成 JWT（含用户ID、角色）
    \item \textbf{请求：} 前端将 Token 放入 Authorization 请求头
    \item \textbf{校验：} 后端 JWT 过滤器校验 Token 合法性
    \item \textbf{刷新：} Access Token 过期后使用 Refresh Token 获取新 Token
    \item \textbf{登出：} 前端清除本地 Token（可选：后端将 Token 加入黑名单）
\end{enumerate}

\subsection{Token 有效期}

\begin{longtable}{p{0.2\textwidth}|p{0.3\textwidth}|p{0.35\textwidth}}
    \toprule
    \textbf{Token 类型} & \textbf{有效期} & \textbf{说明} \\
    \midrule
    \endhead
    \bottomrule
    \endfoot

    Access Token & 2 小时 & 正常接口访问 \\
    Refresh Token & 7 天 & 刷新 Access Token \\
\end{longtable}

\subsection{安全防护}

\begin{itemize}[itemsep=3pt]
    \item \textbf{密码安全：} BCrypt 加密存储，禁止明文
    \item \textbf{接口防刷：} Redis 实现 IP + 接口级别频率限制
    \item \textbf{XSS 防护：} 输入转义 + 富文本过滤
    \item \textbf{SQL 注入防护：} MyBatis-Plus 参数绑定
    \item \textbf{数据脱敏：} 手机号中间4位 * 隐藏
\end{itemize}
