% ==================== 5. 优化建议 ====================
\section{SQL 优化建议}

\subsection{索引优化}

\begin{itemize}[itemsep=3pt]
    \item 高频查询字段（status、created\_at）建立索引
    \item 组合查询条件建立联合索引，例如 (status, created\_at)
    \item 商品表建立全文索引用于搜索
    \item 定期检查并删除冗余索引
\end{itemize}

\subsection{查询优化}

\begin{itemize}[itemsep=3pt]
    \item 避免 SELECT *，只查询需要的字段
    \item 使用 EXPLAIN 分析执行计划，确保走索引
    \item 大分页查询使用覆盖索引 + 子查询优化
    \item 复杂统计使用缓存或异步计算
\end{itemize}

\subsection{数据库维护}

\begin{itemize}[itemsep=3pt]
    \item 定期执行 OPTIMIZE TABLE 回收碎片
    \item 监控慢查询日志，优化慢 SQL
    \item 每日自动备份，保留最近7天
\end{itemize}
